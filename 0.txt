  useEffect(() => {

  }, []);

  const fetchWithRetry = async (url, options, retries = 5, backoff = 1000) => {
    try {
      const response = await fetch(url, options);
      if (!response.ok) {
        const errText = await response.text();
        throw new Error(`HTTP ${response.status}: ${errText}`);
      }
      return await response.json();
    } catch (error) {
      if (retries > 0) {
        await new Promise(resolve => setTimeout(resolve, backoff));
        return fetchWithRetry(url, options, retries - 1, backoff * 2);
      }
      throw error;
    }
  };

  const generateImage = async (promptText) => {
    setLoading(true);
    setErrorMessage("");
    try {
      const url = `https://generativelanguage.googleapis.com/v1beta/models/imagen-4.0-generate-001:predict?key=${apiKey}`;
      // 加入 no text 指令以避免生成任何標題
      const payload = {
        instances: [{ prompt: `Educational cartoon for kids, simple and clean illustration, NO TEXT in image, NO LABELS: ${promptText}` }],
        parameters: { sampleCount: 1 }
      };

      const result = await fetchWithRetry(url, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(payload)
      });

      if (result.predictions?.[0]?.bytesBase64Encoded) {
        setGeneratedImg(`data:image/png;base64,${result.predictions[0].bytesBase64Encoded}`);
      }
    } catch (error) {

      setErrorMessage("圖片載入失敗，請點擊「下一題」或重新整理。");
    } finally {
      setLoading(false);

    const userPrompt = `設計一個關於「${randomTopic}」的教材。`;

    try {
      const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`;
      
      const payload = {
        contents: [{ parts: [{ text: userPrompt }] }],
        systemInstruction: { parts: [{ text: systemPrompt }] },
        generationConfig: { 
          responseMimeType: "application/json",
          responseSchema: {
            type: "OBJECT",
            properties: {
              title: { type: "STRING" },
              scenario: { type: "STRING" },
              prompts: { type: "ARRAY", items: { type: "STRING" }, minItems: 2, maxItems: 2 },
              answer: { type: "STRING" }
            },
            required: ["title", "scenario", "prompts", "answer"]
          }
        }
      };

      const result = await fetchWithRetry(url, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(payload)
      });

      const text = result.candidates?.[0]?.content?.parts?.[0]?.text;
      if (text) {
        const newQuestion = JSON.parse(text);
        setQuestions(prev => [newQuestion, ...prev]);
        setCurrentIdx(0);
        setShowAnswer(false);
        await generateImage(newQuestion.scenario);
      }
    } catch (error) {
      setErrorMessage("AI 生成忙碌中，已切換至預設題庫。");

      setLoading(false);

          <div className="lg:col-span-7">
            <section className="bg-white p-6 rounded-3xl shadow-sm border border-slate-100 h-full flex flex-col">
              <div className="flex items-center justify-between mb-4">
                <h2 className="text-xl font-bold text-indigo-700 flex items-center gap-2">
                  <ImageIcon className="w-6 h-6" />
                  情境圖片
                </h2>

                ) : (

                  </div>
                )}
              </div>

          </div>

          <div className="lg:col-span-5 space-y-6 flex flex-col">

                看圖說說看
              </h2>
              <div className="space-y-4">
                {currentQ.prompts.map((prompt, idx) => (

                ))}
              </div>
            </section>

              <div className="flex justify-between items-center mb-4">
                <h2 className="text-lg font-bold text-amber-600 flex items-center gap-2">
                  <Eye className="w-6 h-6" />
                  參考解答
                </h2>
                <button 
                  onClick={() => setShowAnswer(!showAnswer)} 
                  className={`text-sm px-3 py-1 rounded-full font-medium transition-colors ${showAnswer ? "bg-slate-100 text-slate-500" : "bg-amber-100 text-amber-700"}`}
                >
                  {showAnswer ? "隱藏" : "顯示"}
                </button>
              </div>
              <div 
                className={`p-6 rounded-2xl cursor-pointer min-h-[150px] flex items-center justify-center border-2 transition-all duration-300 ${
                  showAnswer 
                  ? "bg-amber-50 border-amber-200 shadow-inner" 
                  : "bg-slate-50 border-dashed border-slate-200 hover:bg-slate-100"
                }`}
                onClick={() => setShowAnswer(true)}
              >
                {showAnswer ? (
                  <p className="text-slate-800 font-medium text-lg leading-relaxed animate-in fade-in zoom-in-95 duration-300">
                    {currentQ.answer}
                  </p>
                ) : (
                  <div className="text-slate-400 flex flex-col items-center gap-3">
                    <EyeOff className="w-10 h-10 opacity-10"/>
                    <span className="font-bold text-lg">點擊揭曉參考答案</span>
                  </div>
                )}
              </div>
            </section>

            <button 
              onClick={nextQuestion} 
              disabled={loading}
              className="w-full bg-indigo-600 hover:bg-indigo-700 text-white py-5 rounded-3xl font-bold text-xl flex justify-center items-center gap-3 shadow-xl shadow-indigo-200 transition-all active:scale-95"
            >
              下一題 <ChevronRight className="w-7 h-7" />
            </button>
          </div>
        </div>
      </main>
    </div>
  );
};
