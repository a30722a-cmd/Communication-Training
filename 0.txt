        async function fetchWithRetry(url, options, retries = 5) {
            for (let i = 0; i < retries; i++) {
                try {
                    const response = await fetch(url, options);
                    if (response.ok) return await response.json();
                    if (response.status === 429) {
                        await new Promise(res => setTimeout(res, Math.pow(2, i) * 1000));
                        continue;
                    }
                    throw new Error("API Failed");
                } catch (e) {
                    if (i === retries - 1) throw e;
                    await new Promise(res => setTimeout(res, Math.pow(2, i) * 1000));
                }
            }
        }

        async function speakScenario() {
            const text = document.getElementById('scenario-text').innerText;
            const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-tts:generateContent?key=${apiKey}`;
            try {
                const response = await fetchWithRetry(url, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({
                        contents: [{ parts: [{ text: `Say playfully: ${text}` }] }],
                        generationConfig: {
                            responseModalities: ["AUDIO"],
                            speechConfig: { voiceConfig: { prebuiltVoiceConfig: { voiceName: "Puck" } } }
                        }
                    })
                });
                const base64Audio = response.candidates?.[0]?.content?.parts?.find(p => p.inlineData)?.inlineData?.data;
                if (base64Audio) {
                    const audioBlob = pcmToWav(base64Audio, 24000);
                    const audio = new Audio(URL.createObjectURL(audioBlob));
                    audio.play();
                }
            } catch (e) {
                const msg = new SpeechSynthesisUtterance(text);
                msg.lang = 'zh-TW';
                window.speechSynthesis.speak(msg);
            }
        }

        function pcmToWav(base64Data, sampleRate) {
            const byteCharacters = atob(base64Data);
            const buffer = new ArrayBuffer(44 + byteCharacters.length);
            const view = new DataView(buffer);
            const writeString = (o, s) => { for (let i=0; i<s.length; i++) view.setUint8(o+i, s.charCodeAt(i)); };
            writeString(0, 'RIFF');
            view.setUint32(4, 36 + byteCharacters.length, true);
            writeString(8, 'WAVE');
            writeString(12, 'fmt ');
            view.setUint32(16, 16, true);
            view.setUint16(20, 1, true);
            view.setUint16(22, 1, true);
            view.setUint32(24, sampleRate, true);
            view.setUint32(28, sampleRate * 2, true);
            view.setUint16(32, 2, true);
            view.setUint16(34, 16, true);
            writeString(36, 'data');
            view.setUint32(40, byteCharacters.length, true);
            for (let i=0; i<byteCharacters.length; i++) view.setUint8(44+i, byteCharacters.charCodeAt(i));
            return new Blob([view], { type: 'audio/wav' });
        }

                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({
                        instances: { prompt: `Colorful storybook illustration of ${promptText}, expressive characters, clean background.` },
                        parameters: { sampleCount: 1 }
                    })
                });
                if (result.predictions?.[0]) imgEl.src = `data:image/png;base64,${result.predictions[0].bytesBase64Encoded}`;
            } catch (error) { imgEl.src = `https://via.placeholder.com/800x600?text=Social+Adventure`; } finally {
                loadingEl.classList.add('hidden');
                imgEl.style.opacity = '1';
            }
        }

        function renderEmotions() {
            ['my-emotions', 'others-emotions'].forEach(id => {
                const container = document.getElementById(id);
                container.innerHTML = '';
                emotions.forEach(emo => {
                    const btn = document.createElement('button');

                    btn.innerHTML = `<span class="text-3xl mb-1">${emo.emoji}</span><span class="text-xs font-black uppercase">${emo.name}</span>`;

                    };
                    container.appendChild(btn);
                });
            });
        }
